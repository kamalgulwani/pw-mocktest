#include<iostream>
using namespace std;
int f(string str){
for(int i=0;i<str.size();i++){
    for(int j=i;j<str.size();j++){
       if(str[j]!=str[i]){
        continue;
       }
    else{ break;}
    }
    return i;
}
return -1;
}
int main(){
    string str="jhvkjvivivu";
    cout<<f(str);
    return 0;
}
